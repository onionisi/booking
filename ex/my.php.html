function show_cartnum(){
b = ($.cookie('cartn') == null || $.cookie('cartn') == '') ? 0 : $.cookie('cartn');
var p=new RegExp("[0-9]{1,}");
b = parseInt(p.exec(b));

$("#cart_num").toggle(b > 0);
if (b > 0){
$("#cart_num").show()
}else{
$("#cart_num").hide()
}
if (b > 9) {
$("#cart_num").html('N');
} else {
$("#cart_num").html(b);
}
}
$(document).ready(function(){
show_cartnum();
})
